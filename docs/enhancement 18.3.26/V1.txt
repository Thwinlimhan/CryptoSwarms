# UPGRADE 1 — MiroFish: Emergent Swarm Market Simulation

## Purpose

Replace the single static `DefaultRegimeTagger` with a market microstructure simulator that
runs **thousands of personality-driven trader agents** (momentum chasers, mean-reverters,
noise traders, whales) and derives emergent regime signals from their aggregate behavior.
Gate 7 will refuse to promote any candidate that the simulated swarm collectively fades.

```python
"""MiroFish emergent swarm regime simulator.

Calls the MiroFish REST API to run a multi-agent market simulation and returns
a structured regime verdict with agent consensus metrics.
"""
from __future__ import annotations

import statistics
from dataclasses import dataclass
from typing import Any, Protocol


class MiroFishTransport(Protocol):
    def post(self, url: str, payload: dict[str, Any]) -> dict[str, Any]: ...


@dataclass(frozen=True)
class SwarmRegimeVerdict:
    regime: str                   # "bullish" | "bearish" | "choppy" | "breakout"
    consensus_score: float        # 0.0–1.0, fraction of agents agreeing on regime
    momentum_agents_pct: float
    mean_revert_agents_pct: float
    noise_agents_pct: float
    whale_pressure: float         # net buy pressure from large-cap agents
    raw_response: dict[str, Any]


@dataclass(slots=True)
class MiroFishRegimeSimulator:
    """Wraps the MiroFish microservice to derive emergent regime verdicts.

    Args:
        base_url: URL of the running mirofish container, e.g. "http://mirofish:5001"
        transport: injectable HTTP transport (use requests-based impl in production)
        min_consensus: minimum agent consensus fraction to trust the regime label
        n_agents: number of simulated agents (overrides container default if sent)
    """
    base_url: str
    transport: MiroFishTransport
    min_consensus: float = 0.55
    n_agents: int = 2000

    def simulate(self, market_data: dict[str, Any]) -> SwarmRegimeVerdict:
        """Run a full swarm simulation against the supplied OHLCV market_data dict.

        market_data must contain at minimum:
          - "close": list[float]
          - "volume": list[float]
          - "high": list[float]   (optional but improves accuracy)
          - "low": list[float]    (optional but improves accuracy)
        """
        payload = {
            "candles": {
                "close": market_data.get("close", []),
                "volume": market_data.get("volume", []),
                "high": market_data.get("high", []),
                "low": market_data.get("low", []),
            },
            "n_agents": self.n_agents,
            "personality_mix": {
                "momentum": 0.35,
                "mean_revert": 0.30,
                "noise": 0.25,
                "whale": 0.10,
            },
        }
        resp = self.transport.post(
            self.base_url.rstrip("/") + "/simulate",
            payload,
        )
        return self._parse(resp)

    def _parse(self, resp: dict[str, Any]) -> SwarmRegimeVerdict:
        agents = resp.get("agent_breakdown", {})
        total = max(sum(agents.values()), 1)
        return SwarmRegimeVerdict(
            regime=resp.get("regime", "unknown"),
            consensus_score=float(resp.get("consensus_score", 0.0)),
            momentum_agents_pct=agents.get("momentum", 0) / total,
            mean_revert_agents_pct=agents.get("mean_revert", 0) / total,
            noise_agents_pct=agents.get("noise", 0) / total,
            whale_pressure=float(resp.get("whale_net_pressure", 0.0)),
            raw_response=resp,
        )

    def is_tradeable(self, verdict: SwarmRegimeVerdict) -> bool:
        """Return True if the swarm consensus is strong enough to trade."""
        return (
            verdict.consensus_score >= self.min_consensus
            and verdict.regime in ("bullish", "bearish", "breakout")
        )
```

```python
from agents.backtest.mirofish_simulator import MiroFishRegimeSimulator


def gate_7_swarm_regime(
    candidate: StrategyCandidate,
    simulator: MiroFishRegimeSimulator,
    min_consensus: float = 0.55,
) -> GateResult:
    """Gate 7 — MiroFish emergent swarm consensus check.

    Runs thousands of personality-driven simulated traders against the
    candidate's market_data. Fails candidates the swarm collectively fades.

    Args:
        candidate: strategy candidate with market_data dict
        simulator: configured MiroFishRegimeSimulator instance
        min_consensus: minimum agent agreement fraction (default 0.55)
    """
    try:
        verdict = simulator.simulate(
            candidate.market_data if isinstance(candidate.market_data, dict) else {}
        )
    except Exception as exc:
        return GateResult(
            gate_number=7,
            gate_name="swarm_regime_consensus",
            status=GateStatus.ERROR,
            score=0.0,
            details={"error": str(exc)},
        )

    tradeable = simulator.is_tradeable(verdict)
    return GateResult(
        gate_number=7,
        gate_name="swarm_regime_consensus",
        status=GateStatus.PASS if tradeable else GateStatus.FAIL,
        score=verdict.consensus_score,
        details={
            "regime": verdict.regime,
            "consensus_score": verdict.consensus_score,
            "whale_pressure": verdict.whale_pressure,
            "momentum_pct": verdict.momentum_agents_pct,
            "mean_revert_pct": verdict.mean_revert_agents_pct,
            "noise_pct": verdict.noise_agents_pct,
            "min_consensus_threshold": min_consensus,
        },
    )
```

# UPGRADE 2 — Karpathy Autoresearch: Full Code-Level Autonomous Evolution

## Purpose

Upgrade `AutoResearchRunner` from a **parameter-only** optimizer to a **full code editor**.
Every overnight experiment now: reads the current strategy `.py` source → asks Claude/GPT
to propose a code mutation → writes the mutation to disk → runs `ValidationPipeline` →
keeps winners, reverts losers. This is Karpathy-style “AI scientist” applied to trading.

## Edit — `agents/evolution/autoresearch.py`

Replace the current `AutoResearchRunner.run()` body with the Karpathy loop:

```python
"""
FULL REPLACEMENT for agents/evolution/autoresearch.py
Adds: CodeEditorProtocol, CodeMutation, KarpathyAutoResearchRunner
Keeps: AutoResearchPolicy, ExperimentDecision, AutoResearchReport, load_program_policy
"""
from __future__ import annotations

import ast
import shutil
import tempfile
from dataclasses import dataclass
from datetime import datetime, timedelta, timezone
from pathlib import Path
from typing import Callable, Protocol

from agents.evolution.deap_nightly import NightlyOptimizer, OptimizationCandidate


# ── Existing unchanged ──────────────────────────────────────────────────────
# (keep AutoResearchPolicy, ExperimentDecision, AutoResearchReport,
#  load_program_policy, _as_utc exactly as-is)


# ── Code-level evolution additions ─────────────────────────────────────────

class CodeEditorProtocol(Protocol):
    """Adapter for any LLM-backed code mutation service (Claude, GPT-4, local)."""
    def propose_mutation(
        self,
        source_code: str,
        score_before: float,
        failure_hint: str | None,
    ) -> str:
        """Return mutated Python source code string. Must be syntactically valid."""
        ...


@dataclass(frozen=True)
class CodeMutation:
    strategy_path: Path
    original_source: str
    mutated_source: str
    proposed_by: str
    failure_hint: str | None


def _validate_syntax(source: str) -> bool:
    try:
        ast.parse(source)
        return True
    except SyntaxError:
        return False


def _apply_mutation(mutation: CodeMutation) -> Path:
    """Write mutated source to a temp file. Caller is responsible for cleanup."""
    tmp = tempfile.NamedTemporaryFile(
        suffix=".py",
        dir=mutation.strategy_path.parent,
        delete=False,
        mode="w",
        encoding="utf-8",
    )
    tmp.write(mutation.mutated_source)
    tmp.close()
    return Path(tmp.name)


def _revert_mutation(tmp_path: Path) -> None:
    try:
        tmp_path.unlink(missing_ok=True)
    except OSError:
        pass


@dataclass(slots=True)
class KarpathyAutoResearchRunner:
    """Code-editing autonomous research runner.

    For each experiment:
      1. Reads current strategy .py source
      2. Asks code_editor to propose a mutation (with optional failure_hint)
      3. Validates syntax
      4. Writes mutation to a temp .py file
      5. Runs ValidationPipeline (or score_fn proxy) on mutated file
      6. Keeps winners (overwrites original), reverts losers

    Args:
        strategy_path: path to the strategy .py file to evolve
        score_fn: callable(params) -> float used for fast scoring
        base_params: seed parameter dict passed to NightlyOptimizer
        policy: AutoResearchPolicy controlling runtime limits
        code_editor: LLM-backed CodeEditorProtocol instance
        chub_client: optional ContextHubClient to fetch fresh API docs before editing
        slippage: slippage fraction passed to NightlyOptimizer
    """
    strategy_path: Path
    score_fn: Callable[[dict[str, float]], float]
    base_params: dict[str, float]
    policy: "AutoResearchPolicy"
    code_editor: CodeEditorProtocol
    chub_client: object | None = None   # ChubClient — see Upgrade 5
    slippage: float = 0.004

    def run(
        self,
        incumbent_score: float,
        now_provider: Callable[[], datetime] | None = None,
    ) -> "AutoResearchReport":
        now_fn = now_provider or (lambda: datetime.now(timezone.utc))
        started_at = _as_utc(now_fn())
        deadline = started_at + self.policy.max_runtime
        kept: list["ExperimentDecision"] = []
        discarded: list["ExperimentDecision"] = []
        incumbent = float(incumbent_score)
        experiments = 0
        failure_hint: str | None = None

        for experiment_id in range(1, self.policy.max_experiments + 1):
            if _as_utc(now_fn()) >= deadline:
                break

            # ── Step 1: optionally fetch fresh API docs via Chub ───────────
            if self.chub_client is not None:
                try:
                    self.chub_client.refresh_strategy_api_context(self.strategy_path)
                except Exception:
                    pass  # non-blocking; log in production

            # ── Step 2: read source ─────────────────────────────────────────
            original_source = self.strategy_path.read_text(encoding="utf-8")

            # ── Step 3: LLM proposes mutation ───────────────────────────────
            try:
                mutated_source = self.code_editor.propose_mutation(
                    source_code=original_source,
                    score_before=incumbent,
                    failure_hint=failure_hint,
                )
            except Exception as exc:
                discarded.append(ExperimentDecision(
                    experiment_id=experiment_id,
                    best_score=incumbent,
                    incumbent_before=incumbent,
                    score_improvement=0.0,
                    kept=False,
                    reason=f"editor_error:{exc}",
                    candidate=None,
                ))
                continue

            # ── Step 4: syntax gate ─────────────────────────────────────────
            if not _validate_syntax(mutated_source):
                discarded.append(ExperimentDecision(
                    experiment_id=experiment_id,
                    best_score=incumbent,
                    incumbent_before=incumbent,
                    score_improvement=0.0,
                    kept=False,
                    reason="syntax_error",
                    candidate=None,
                ))
                failure_hint = "Your last mutation produced a SyntaxError. Fix it."
                continue

            mutation = CodeMutation(
                strategy_path=self.strategy_path,
                original_source=original_source,
                mutated_source=mutated_source,
                proposed_by="code_editor",
                failure_hint=failure_hint,
            )
            tmp_path = _apply_mutation(mutation)

            # ── Step 5: score mutated strategy ─────────────────────────────
            try:
                optimizer = NightlyOptimizer(
                    score_fn=self.score_fn,
                    base_params=self.base_params,
                    slippage=self.slippage,
                )
                candidates = optimizer.run(
                    generations=self.policy.generations_per_experiment,
                    step=self.policy.mutation_step,
                )
                experiments += 1
                top_score = max((c.score for c in candidates), default=0.0)
                top_candidate = max(candidates, key=lambda c: c.score) if candidates else None
            except Exception as exc:
                _revert_mutation(tmp_path)
                discarded.append(ExperimentDecision(
                    experiment_id=experiment_id,
                    best_score=incumbent,
                    incumbent_before=incumbent,
                    score_improvement=0.0,
                    kept=False,
                    reason=f"scoring_error:{exc}",
                    candidate=None,
                ))
                failure_hint = f"Scoring failed: {exc}. Revise the strategy logic."
                continue

            delta = top_score - incumbent
            keep = delta >= self.policy.min_score_improvement

            # ── Step 6: promote or revert ───────────────────────────────────
            if keep:
                shutil.copy2(tmp_path, self.strategy_path)
                incumbent = top_score
                failure_hint = None
                kept.append(ExperimentDecision(
                    experiment_id=experiment_id,
                    best_score=top_score,
                    incumbent_before=float(incumbent_score) if experiment_id == 1 else kept[-1].best_score,
                    score_improvement=round(delta, 6),
                    kept=True,
                    reason="promoted",
                    candidate=top_candidate,
                ))
            else:
                failure_hint = (
                    f"Score improved only {delta:.4f} (need {self.policy.min_score_improvement}). "
                    "Try a more aggressive structural change."
                )
                discarded.append(ExperimentDecision(
                    experiment_id=experiment_id,
                    best_score=top_score,
                    incumbent_before=incumbent,
                    score_improvement=round(delta, 6),
                    kept=False,
                    reason="below_threshold",
                    candidate=None,
                ))
            _revert_mutation(tmp_path)

        kept = sorted(kept, key=lambda d: d.best_score, reverse=True)[: self.policy.keep_top_k]
        return AutoResearchReport(
            started_at=started_at,
            finished_at=_as_utc(now_fn()),
            experiments_run=experiments,
            final_incumbent_score=round(incumbent, 6),
            kept=kept,
            discarded=discarded,
        )
```

```python
"""Senpi AI Hyperliquid smart-money feed connector.

Connects to the Senpi AI API to receive aggregated smart-money flow signals
from Hyperliquid perps. Returns normalized directional bias and divergence score.
"""
from __future__ import annotations

from dataclasses import dataclass
from typing import Any, Protocol


class SenpiTransport(Protocol):
    def get(self, url: str, params: dict[str, Any]) -> dict[str, Any]: ...


@dataclass(frozen=True)
class SmartMoneySignal:
    symbol: str
    direction: str          # "long" | "short" | "neutral"
    confidence: float       # 0.0–1.0
    net_flow_usd: float     # positive = net buy, negative = net sell
    funding_divergence: float   # spot vs perp funding diff (bps)
    large_print_detected: bool
    raw: dict[str, Any]


@dataclass(slots=True)
class SenpiHyperfeedConnector:
    """Fetches smart-money signals from the Senpi AI REST API.

    Args:
        base_url: Senpi API endpoint, e.g. "https://api.senpi.ai/v1"
        api_key: your Senpi API key (set via SENPI_API_KEY env var)
        transport: injectable HTTP transport
        large_print_threshold_usd: USD threshold to flag a large institutional print
    """
    base_url: str
    api_key: str
    transport: SenpiTransport
    large_print_threshold_usd: float = 500_000.0

    def fetch_signal(self, symbol: str, lookback_minutes: int = 60) -> SmartMoneySignal:
        resp = self.transport.get(
            self.base_url.rstrip("/") + "/hyperliquid/smart-money",
            {
                "symbol": symbol,
                "lookback_minutes": lookback_minutes,
                "api_key": self.api_key,
            },
        )
        net_flow = float(resp.get("net_flow_usd", 0.0))
        confidence = float(resp.get("confidence", 0.0))
        funding_div = float(resp.get("funding_divergence_bps", 0.0))
        large_print = abs(net_flow) >= self.large_print_threshold_usd

        if confidence < 0.4 or abs(net_flow) < 50_000:
            direction = "neutral"
        elif net_flow > 0:
            direction = "long"
        else:
            direction = "short"

        return SmartMoneySignal(
            symbol=symbol,
            direction=direction,
            confidence=confidence,
            net_flow_usd=net_flow,
            funding_divergence=funding_div,
            large_print_detected=large_print,
            raw=resp,
        )

    def fetch_inverted_sentinel(self, symbol: str) -> SmartMoneySignal:
        """Inverted scan: returns the OPPOSITE of detected smart-money direction.

        Use this to fade retail-dominated moves where smart money is absent.
        """
        signal = self.fetch_signal(symbol)
        inverted_direction = {
            "long": "short",
            "short": "long",
            "neutral": "neutral",
        }[signal.direction]
        return SmartMoneySignal(
            symbol=signal.symbol,
            direction=inverted_direction,
            confidence=signal.confidence,
            net_flow_usd=-signal.net_flow_usd,
            funding_divergence=signal.funding_divergence,
            large_print_detected=signal.large_print_detected,
            raw=signal.raw,
        )
```

```python
# ── Gate 7/8: Smart-Money Divergence Gate ──────────────────────────────────
# Add import at top of gates.py:
# from agents.research.SenpiHyperfeedConnector import SenpiHyperfeedConnector, SmartMoneySignal

def gate_7_smart_money_divergence(
    candidate: StrategyCandidate,
    senpi_connector: "SenpiHyperfeedConnector",
    max_divergence_confidence: float = 0.65,
) -> GateResult:
    """Gate 7 — Senpi AI smart-money anti-divergence check.

    Fails the candidate if Senpi detects high-confidence smart money
    flowing in the OPPOSITE direction to the strategy's signal.

    Args:
        candidate: must have candidate.params["signal_direction"] in {"long","short"}
        senpi_connector: configured SenpiHyperfeedConnector
        max_divergence_confidence: if smart money diverges above this, FAIL
    """
    symbol = candidate.params.get("symbol", "BTC-PERP")
    strategy_direction = str(candidate.params.get("signal_direction", "long"))

    try:
        signal = senpi_connector.fetch_signal(str(symbol))
    except Exception as exc:
        return GateResult(
            gate_number=7,
            gate_name="smart_money_divergence",
            status=GateStatus.ERROR,
            score=0.0,
            details={"error": str(exc)},
        )

    diverging = (
        signal.direction != "neutral"
        and signal.direction != strategy_direction
        and signal.confidence >= max_divergence_confidence
    )

    return GateResult(
        gate_number=7,
        gate_name="smart_money_divergence",
        status=GateStatus.FAIL if diverging else GateStatus.PASS,
        score=1.0 - signal.confidence if diverging else signal.confidence,
        details={
            "strategy_direction": strategy_direction,
```

```python
# ── Senpi Hardened DSL Stop Validator ──────────────────────────────────────

import re as _re

_DSL_PATTERN = _re.compile(
    r"^(?:entry[_\s]*[+\-\*\/]\s*[\d.]+|atr\(\d+\)\s*[+\-\*]\s*[\d.]+|fixed\([\d.]+\))$",
    _re.IGNORECASE,
)

_DSL_BUILTINS: dict[str, object] = {
    "atr": lambda n: 0.0,   # replaced at runtime with real ATR(n) value
    "entry": 0.0,
    "fixed": lambda x: float(x),
}


class DSLValidationError(ValueError):
    """Raised when a stop/TP expression does not match the hardened DSL."""


def validate_dsl_stop(expression: str) -> str:
    """Validate a stop-loss or take-profit DSL expression.

    Allowed forms:
      - "entry - 0.02"         (entry-relative fixed offset)
      - "entry * 0.98"         (entry-relative percentage)
      - "atr(14) * 1.5"        (ATR-multiple)
      - "fixed(42000.5)"       (absolute price)

    Raises DSLValidationError if expression is invalid.
    Returns the stripped expression if valid.
    """
    expr = expression.strip()
    if not _DSL_PATTERN.match(expr):
        raise DSLValidationError(
            f"Invalid stop DSL: '{expr}'. "
            "Use: 'entry ± N', 'entry * N', 'atr(N) * M', or 'fixed(P)'"
        )
    return expr


def evaluate_dsl_stop(expression: str, *, entry_price: float, atr_value: float) -> float:
    """Evaluate a validated DSL stop expression to a concrete float price.

    Args:
        expression: validated DSL string from validate_dsl_stop()
        entry_price: actual fill price
        atr_value: current ATR(N) value for atr() expressions
    """
    builtins = {
        "atr": lambda n: atr_value,
        "entry": entry_price,
        "fixed": lambda x: float(x),
        "__builtins__": {},
    }
    try:
        result = eval(expression, {"__builtins__": {}}, builtins)  # noqa: S307
        return float(result)
    except Exception as exc:
        raise DSLValidationError(f"DSL evaluation failed: {exc}") from exc
```

```python
"""Context Hub (Chub) client — zero-hallucination API documentation fetcher.

Wraps the local Chub server to retrieve live, structured API documentation
before any LLM-generated code edit or external API call.
"""
from __future__ import annotations

from dataclasses import dataclass
from pathlib import Path
from typing import Any, Protocol


class ChubTransport(Protocol):
    def get(self, url: str, params: dict[str, Any]) -> dict[str, Any]: ...


@dataclass(frozen=True)
class ApiDoc:
    tool_name: str
    version: str
    endpoints: list[dict[str, Any]]
    schemas: dict[str, Any]
    raw: dict[str, Any]


@dataclass(slots=True)
class ChubClient:
    """Fetches live API documentation from the local Chub server.

    Args:
        base_url: Chub server URL, e.g. "http://chub:7777"
        transport: injectable HTTP transport
        cache_ttl_seconds: how long to cache docs locally (default 300s)
    """
    base_url: str
    transport: ChubTransport
    cache_ttl_seconds: int = 300
    _cache: dict[str, tuple[float, ApiDoc]] = None  # type: ignore

    def __post_init__(self) -> None:
        object.__setattr__(self, "_cache", {})

    def get_api_doc(self, tool_name: str) -> ApiDoc:
        """Fetch documentation for a named tool/API.

        Results are cached for cache_ttl_seconds to avoid hammering Chub.
        """
        import time
        now = time.monotonic()
        if tool_name in self._cache:
            ts, doc = self._cache[tool_name]
            if now - ts < self.cache_ttl_seconds:
                return doc

        resp = self.transport.get(
            self.base_url.rstrip("/") + "/docs",
            {"tool": tool_name},
        )
        doc = ApiDoc(
            tool_name=tool_name,
            version=resp.get("version", "unknown"),
            endpoints=resp.get("endpoints", []),
            schemas=resp.get("schemas", {}),
            raw=resp,
        )
        self._cache[tool_name] = (now, doc)
        return doc

    def refresh_strategy_api_context(self, strategy_path: Path) -> str:
        """Extract tool names referenced in strategy_path and return concatenated docs.

        Scans the file for known tool imports (jesse, vectorbt, hyperliquid) and
        prefetches their docs, returning them as a formatted context string for the LLM.
        """
        source = strategy_path.read_text(encoding="utf-8")
        tools_detected = []
        for known_tool in ("jesse", "vectorbt", "hyperliquid", "binance", "okx", "senpi"):
            if known_tool in source.lower():
                tools_detected.append(known_tool)

        context_parts = []
        for tool in tools_detected:
            try:
                doc = self.get_api_doc(tool)
                context_parts.append(
                    f"## {doc.tool_name} API v{doc.version}\n"
                    + "\n".join(
                        f"  {ep.get('method','GET')} {ep.get('path','')} — {ep.get('summary','')}"
                        for ep in doc.endpoints[:10]
                    )
                )
            except Exception:
                pass
        return "\n\n".join(context_parts)

    def build_gate_api_context(self) -> str:
        """Fetch docs for all gate-related APIs and return as LLM context block."""
        return self.refresh_strategy_api_context(Path("agents/backtest/gates.py"))
```

```python
    """Fetches funding rate predictions from ORMR Boros + Synth AI.

    Args:
        boros_url: ORMR Boros API base URL
        synth_url: Synth AI API base URL
        api_key: shared API key for both services
        transport: injectable HTTP transport

    synth_url: str

            self.synth_url.rstrip("/") + "/funding/predict",

        )

            self.boros_url.rstrip("/") + "/yield/opportunity",
            {"symbol": symbol, "venue": venue, "api_key": self.api_key},
        )

        current_rate = float(synth_resp.get("current_funding_bps", 0.0))
        predicted_rate = float(synth_resp.get("predicted_funding_bps", 0.0))
        flip_minutes = synth_resp.get("flip_in_minutes")

        return FundingPrediction(
            symbol=symbol,
            current_funding_rate=current_rate,
            predicted_funding_rate=predicted_rate,
            predicted_flip_in_minutes=int(flip_minutes) if flip_minutes is not None else None,
            confidence=float(synth_resp.get("confidence", 0.0)),
            yield_opportunity_score=float(boros_resp.get("yield_score", 0.0)),
            venue=venue,
            raw={"synth": synth_resp, "boros": boros_resp},
        )

    def scan_opportunities(
        self,
        symbols: list[str],
        venues: list[str] | None = None,
    ) -> list[FundingPrediction]:
        """Scan multiple symbols/venues and return high-yield predictions sorted by score."""
        venues = venues or ["hyperliquid", "binance", "bybit"]
        results = []
        for symbol in symbols:
            for venue in venues:
                try:
                    pred = self.fetch_funding_prediction(symbol, venue)
                    if pred.yield_opportunity_score >= self.min_yield_score:
                        results.append(pred)
                except Exception:
                    continue
        return sorted(results, key=lambda p: p.yield_opportunity_score, reverse=True)
```

```python
@dataclass(slots=True)
class BorosYieldUnitRunner:

    For each FundingPrediction, simulates a delta-neutral yield trade:
    long spot + short perp when funding is positive (collect funding),
    or short spot + long perp when funding is negative (pay the other side).

    Returns a list[float] of per-period returns compatible with ValidationPipeline.
    """
    connector: ORMRResearchConnector

    def run_yield_backtest(
        self,
        symbol: str,
        funding_history: list[float],  # historical 8h funding rates in bps
        entry_cost_bps: float = 2.0,

        returns: list[float] = []
        for i in range(1, len(funding_history)):
            funding_bps = funding_history[i]
            # Positive funding → short perp, collect; Negative → long perp, collect
            gross = abs(funding_bps) / 10_000  # convert bps to fraction
            net = gross - (entry_cost_bps / 10_000)
            returns.append(max(net, -entry_cost_bps / 10_000))
```

```python
            score=0.0,
            details={"error": str(exc)},

    passed = (
        pred.confidence >= min_confidence
        and pred.yield_opportunity_score >= min_yield_score
    )

    return GateResult(
        gate_number=8,
        gate_name="funding_synth_prediction",
        status=GateStatus.PASS if passed else GateStatus.FAIL,
```

```python
"""HyperspaceAI Prometheus P2P mesh integration.

Replaces single-node RuntimeOrchestrator with a distributed agent mesh.
Nodes gossip MemoryDag entries, validation summaries, and evolution winners.
AgentRank consensus determines global strategy promotion.
"""
from __future__ import annotations

import hashlib
import json
from dataclasses import dataclass, field
from datetime import datetime, timezone
from typing import Any, Protocol


class HyperspaceTransport(Protocol):
    def post(self, url: str, payload: dict[str, Any]) -> dict[str, Any]: ...
    def get(self, url: str, params: dict[str, Any]) -> dict[str, Any]: ...


@dataclass(frozen=True)
class GossipMessage:
    message_id: str
    origin_node_id: str
    message_type: str    # "memory_dag_node" | "validation_result" | "evolution_winner"
    topic: str
    payload: dict[str, Any]
    timestamp: datetime
    signature: str       # sha256(origin+type+topic+payload_json)


@dataclass(frozen=True)
class AgentRankVote:
    strategy_id: str
    voter_node_id: str
    score: float
    gate_results: list[dict[str, Any]]
    timestamp: datetime


@dataclass(slots=True)
class HyperspaceMeshClient:
    """Connects to the local Hyperspace node and gossips intelligence.

    Args:
        base_url: local Hyperspace node API, e.g. "http://hyperspace-node:8080"
        node_id: unique identifier for this CryptoSwarms instance
        transport: injectable HTTP transport
        network: Hyperspace network name
    """
    base_url: str
    node_id: str
    transport: HyperspaceTransport
    network: str = "cryptoswarms-mainnet"

    def _sign(self, origin: str, msg_type: str, topic: str, payload: dict[str, Any]) -> str:
        raw = f"{origin}:{msg_type}:{topic}:{json.dumps(payload, sort_keys=True)}"
        return hashlib.sha256(raw.encode()).hexdigest()[:16]

    def gossip_memory_node(
        self,
        topic: str,
        content: str,
        metadata: dict[str, Any] | None = None,
    ) -> GossipMessage:
        """Broadcast a MemoryDag node to all mesh peers."""
        payload = {"topic": topic, "content": content, "metadata": metadata or {}}
        msg = GossipMessage(
            message_id=self._sign(self.node_id, "memory_dag_node", topic, payload),
            origin_node_id=self.node_id,
            message_type="memory_dag_node",
            topic=topic,
            payload=payload,
            timestamp=datetime.now(timezone.utc),
            signature=self._sign(self.node_id, "memory_dag_node", topic, payload),
        )
        self.transport.post(
            self.base_url.rstrip("/") + "/gossip/broadcast",
            {
                "network": self.network,
                "message": {
                    "message_id": msg.message_id,
                    "origin_node_id": msg.origin_node_id,
                    "message_type": msg.message_type,
                    "topic": msg.topic,
                    "payload": msg.payload,
                    "timestamp": msg.timestamp.isoformat(),
                    "signature": msg.signature,
                },
            },
        )
        return msg

    def gossip_validation_result(
        self,
        strategy_id: str,
        gate_results: list[dict[str, Any]],
        final_score: float,
    ) -> GossipMessage:
        """Broadcast a validation result for AgentRank consensus."""
        payload = {
            "strategy_id": strategy_id,
            "gate_results": gate_results,
            "final_score": final_score,
        }
        msg = GossipMessage(
            message_id=self._sign(self.node_id, "validation_result", strategy_id, payload),
            origin_node_id=self.node_id,
            message_type="validation_result",
            topic=strategy_id,
            payload=payload,
            timestamp=datetime.now(timezone.utc),
            signature=self._sign(self.node_id, "validation_result", strategy_id, payload),
        )
        self.transport.post(
            self.base_url.rstrip("/") + "/gossip/broadcast",
            {"network": self.network, "message": {
                "message_id": msg.message_id,
                "origin_node_id": msg.origin_node_id,
                "message_type": msg.message_type,
                "topic": msg.topic,
                "payload": msg.payload,
                "timestamp": msg.timestamp.isoformat(),
                "signature": msg.signature,
            }},
        )
        return msg

    def submit_agent_rank_vote(
        self,
        strategy_id: str,
        score: float,
        gate_results: list[dict[str, Any]],
    ) -> dict[str, Any]:
        """Cast an AgentRank vote for a strategy's global promotion."""
        vote = {
            "strategy_id": strategy_id,
            "voter_node_id": self.node_id,
            "score": score,
            "gate_results": gate_results,
            "timestamp": datetime.now(timezone.utc).isoformat(),
        }
        return self.transport.post(
            self.base_url.rstrip("/") + "/agentrank/vote",
            {"network": self.network, "vote": vote},
        )

    def get_global_consensus(self, strategy_id: str) -> dict[str, Any]:
        """Fetch AgentRank global consensus for a strategy across all mesh peers."""
        return self.transport.get(
            self.base_url.rstrip("/") + "/agentrank/consensus",
            {"network": self.network, "strategy_id": strategy_id},
        )

    def fetch_peer_memories(self, topic_prefix: str, limit: int = 50) -> list[dict[str, Any]]:
        """Fetch gossip messages from peer nodes for a topic prefix."""
        resp = self.transport.get(
            self.base_url.rstrip("/") + "/gossip/fetch",
            {"network": self.network, "topic_prefix": topic_prefix, "limit": limit},
        )
        return resp.get("messages", [])
```

```python
def hyperspace_runtime_orchestrator(
    mesh_client: "HyperspaceMeshClient | None" = None,
) -> RuntimeOrchestrator:
    """Build a RuntimeOrchestrator with optional Hyperspace mesh gossip middleware."""

    class HyperspaceGossipMiddleware(RuntimeMiddleware):
        name = "hyperspace_gossip"
        def __init__(self, client: "HyperspaceMeshClient") -> None:
            self._client = client
        def run(self, ctx: RuntimeContext, nxt: Callable[[RuntimeContext], RuntimeContext]) -> RuntimeContext:
            result = nxt(ctx)
            try:
                self._client.gossip_memory_node(
                    topic="runtime_checkpoint",
                    content=str(ctx.checkpoints),
                    metadata={"payload_keys": list(ctx.payload.keys())},
                )
            except Exception:
                pass
            return result

    middlewares: list[RuntimeMiddleware] = [
        StageFlagMiddleware("scope", required_flag="scope_ok"),
        StageFlagMiddleware("input_guardrail", required_flag="input_ok"),
        StageFlagMiddleware("tool_guardrail", required_flag="tool_ok"),
        StageFlagMiddleware("debate"),
        StageFlagMiddleware("governor", required_flag="governor_ok"),
    ]
    if mesh_client is not None:
```

```python
# ── Gate 9: Global Swarm Intelligence Gate ─────────────────────────────────

def gate_9_swarm_intelligence(
    candidate: StrategyCandidate,
    mesh_client: "HyperspaceMeshClient",
    min_peer_nodes: int = 3,
    min_consensus_score: float = 0.60,
) -> GateResult:
    """Gate 9 — HyperspaceAI global AgentRank consensus gate.

    Fetches AgentRank consensus from the P2P mesh. Requires at least
    min_peer_nodes independent votes and min_consensus_score average.
    Prevents locally-overfit strategies from going live without global validation.

    Args:
        candidate: strategy candidate with strategy_id
        mesh_client: configured HyperspaceMeshClient
        min_peer_nodes: minimum peer votes required for consensus
        min_consensus_score: minimum average peer score
    """
    try:
        consensus = mesh_client.get_global_consensus(candidate.strategy_id)
    except Exception as exc:
        return GateResult(
            gate_number=9,
            gate_name="swarm_intelligence_consensus",
            status=GateStatus.ERROR,
            score=0.0,
            details={"error": str(exc)},
        )

    vote_count = int(consensus.get("vote_count", 0))
    avg_score = float(consensus.get("average_score", 0.0))
    passed = vote_count >= min_peer_nodes and avg_score >= min_consensus_score

    return GateResult(
        gate_number=9,
        gate_name="swarm_intelligence_consensus",
        status=GateStatus.PASS if passed else GateStatus.FAIL,
        score=avg_score,
        details={
            "vote_count": vote_count,
            "average_score": avg_score,
            "min_peer_nodes": min_peer_nodes,
            "min_consensus_score": min_consensus_score,
            "peer_breakdown": consensus.get("peer_breakdown", []),
        },
    )
```
